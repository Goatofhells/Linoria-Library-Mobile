-- ─── Core helpers ─────────────────────────────────────────────────────────────
function Library:SafeCallback(f, ...)
	if not f then return end
	if not Library.NotifyOnError then return f(...) end
	local ok, err = pcall(f, ...)
	if not ok then
		local _, i = err:find(':%d+: ')
		Library:Notify(i and err:sub(i+1) or err, 3)
	end
end

end

-- ─── Touch tap tracking ───────────────────────────────────────────────────────
Library.TouchTapThreshold = 12
Library.TouchStartPos = nil
Library.TouchIsTap = false

InputService.TouchStarted:Connect(function(Input)
	Library.TouchStartPos = Input.Position
	Library.TouchIsTap = true
end)
InputService.TouchMoved:Connect(function(Input)
	if Library.TouchStartPos then
		if (Input.Position - Library.TouchStartPos).Magnitude > Library.TouchTapThreshold then
			Library.TouchIsTap = false
		end
	end
end)
InputService.TouchEnded:Connect(function()
	Library.TouchStartPos = nil
end)

function Library:IsTap(Input)
	if Input.UserInputType == Enum.UserInputType.MouseButton1 then return true end
	if Input.UserInputType == Enum.UserInputType.Touch then return Library.TouchIsTap end
	return false
end

-- ─── Draggable ────────────────────────────────────────────────────────────────
function Library:MakeDraggable(Instance, Cutoff)
	Instance.Active = true
	Instance.InputBegan:Connect(function(Input)
		if Input.UserInputType == Enum.UserInputType.MouseButton1 then
			local ObjPos = Vector2.new(
				Mouse.X - Instance.AbsolutePosition.X,
				Mouse.Y - Instance.AbsolutePosition.Y)
			if ObjPos.Y > (Cutoff or 40) then return end
			while InputService:IsMouseButtonPressed(Enum.UserInputType.MouseButton1) do
				Instance.Position = UDim2.new(0,
					Mouse.X - ObjPos.X + Instance.Size.X.Offset * Instance.AnchorPoint.X,
					0,
					Mouse.Y - ObjPos.Y + Instance.Size.Y.Offset * Instance.AnchorPoint.Y)
				RenderStepped:Wait()
			end
		end
	end)

	if Library.IsMobile then
		local Dragging, DragInput, DragStart, StartPos
		local DragCommitted = false

		InputService.TouchStarted:Connect(function(Input)
			if Library.CantDragForced then Dragging = false return end
			if not Dragging and Library:MouseIsOverFrame(Instance, Input)
				and Library.Window and Library.Window.Holder.Visible then
				DragInput  = Input
				DragStart  = Input.Position
				StartPos   = Instance.Position
				DragCommitted = false
				local off = Input.Position - DragStart
				if off.Y > (Cutoff or 40) then Dragging = false return end
				Dragging = true
			end
		end)
		InputService.TouchMoved:Connect(function(Input)
			if Library.CantDragForced then Dragging = false DragCommitted = false return end
			if Input == DragInput and Dragging and Library.CanDrag
				and Library.Window and Library.Window.Holder.Visible then
				local off = Input.Position - DragStart
				if not DragCommitted then
					if off.Magnitude < Library.TouchTapThreshold then return end
					DragCommitted = true
				end
				Instance.Position = UDim2.new(
					StartPos.X.Scale, StartPos.X.Offset + off.X,
					StartPos.Y.Scale, StartPos.Y.Offset + off.Y)
			end
		end)
		InputService.TouchEnded:Connect(function(Input)
			if Input == DragInput then Dragging = false DragCommitted = false end
		end)
	end
end

-- ─── Resizable ────────────────────────────────────────────────────────────────
function Library:MakeResizable(Instance, MinSize)
	if Library.IsMobile then return end
	Instance.Active = true
	local SZ = 22
	MinSize = MinSize or Library.MinSize

	local Resizer = Library:Create('Frame', {
		BackgroundTransparency = 1; BorderSizePixel = 0;
		Size = UDim2.fromOffset(SZ, SZ);
		Position = UDim2.new(1, -SZ, 1, -SZ);
		ClipsDescendants = true; ZIndex = 5; Parent = Instance;
	})
	local RBtn = Library:Create('ImageButton', {
		BackgroundColor3 = Library.AccentColor; BackgroundTransparency = 1;
		BorderSizePixel = 0;
		Size = UDim2.new(2,0,2,0); Position = UDim2.new(1,-SZ,1,-SZ);
		ZIndex = 6; Parent = Resizer;
	})
	Library:Create('UICorner', { CornerRadius = UDim.new(0.5,0); Parent = RBtn })
	Library:AddToRegistry(RBtn, { BackgroundColor3 = 'AccentColor' })

	Resizer.Size    = UDim2.fromOffset(SZ, SZ)
	Resizer.Position= UDim2.new(1, -SZ, 1, -SZ)

	local OffsetPos
	local function Finish(t)
		RBtn.Position = UDim2.new(); RBtn.Size = UDim2.new(2,0,2,0)
		RBtn.Parent = Resizer; RBtn.BackgroundTransparency = t; OffsetPos = nil
	end
	RBtn.MouseButton1Down:Connect(function()
		if not OffsetPos then
			OffsetPos = Vector2.new(
				Mouse.X - (Instance.AbsolutePosition.X + Instance.AbsoluteSize.X),
				Mouse.Y - (Instance.AbsolutePosition.Y + Instance.AbsoluteSize.Y))
			RBtn.BackgroundTransparency = 1
			RBtn.Size = UDim2.fromOffset(ScreenGui.AbsoluteSize.X, ScreenGui.AbsoluteSize.Y)
			RBtn.Position = UDim2.new(); RBtn.Parent = ScreenGui
		end
	end)
	RBtn.MouseMoved:Connect(function()
		if OffsetPos then
			local mp = Vector2.new(Mouse.X - OffsetPos.X, Mouse.Y - OffsetPos.Y)
			local fs = Vector2.new(
				math.clamp(mp.X - Instance.AbsolutePosition.X, MinSize.X, math.huge),
				math.clamp(mp.Y - Instance.AbsolutePosition.Y, MinSize.Y, math.huge))
			Instance.Size = UDim2.fromOffset(fs.X, fs.Y)
		end
	end)
	RBtn.MouseEnter:Connect(function() Finish(0.5) end)
	RBtn.MouseLeave:Connect(function() Finish(1) end)
	RBtn.MouseButton1Up:Connect(function() Finish(0.5) end)
end

-- ─── Tooltip ──────────────────────────────────────────────────────────────────
function Library:AddToolTip(Text, Hover)
	local X, Y = Library:GetTextBounds(Text, Library.Font, 13)
	local Tip = Library:Create('Frame', {
		BackgroundColor3 = BG3; BorderColor3 = SEP;
		Size = UDim2.fromOffset(X + 12, Y + 8);
		ZIndex = 200; Visible = false; Parent = ScreenGui;
	})
	Library:Create('Frame', {
		BackgroundColor3 = ACC; BorderSizePixel = 0;
		Size = UDim2.new(0,2,1,0); ZIndex = 201; Parent = Tip;
	})
	Library:CreateLabel({
		Position = UDim2.fromOffset(8, 3);
		Size = UDim2.fromOffset(X, Y);
		TextSize = 13; Text = Text; ZIndex = 201;
		TextColor3 = TXT; TextXAlignment = Enum.TextXAlignment.Left;
		Parent = Tip;
	})
	Library:AddToRegistry(Tip, { BackgroundColor3 = 'MainColor'; BorderColor3 = 'OutlineColor' })
	local Hovering = false
	Hover.MouseEnter:Connect(function()
		if Library:MouseIsOverOpenedFrame() then return end
		Hovering = true
		Tip.Position = UDim2.fromOffset(Mouse.X + 15, Mouse.Y + 12)
		Tip.Visible = true
		while Hovering do
			RunService.Heartbeat:Wait()
			Tip.Position = UDim2.fromOffset(Mouse.X + 15, Mouse.Y + 12)
		end
	end)
	Hover.MouseLeave:Connect(function() Hovering = false Tip.Visible = false end)
	if LibraryMainOuterFrame then
		LibraryMainOuterFrame:GetPropertyChangedSignal('Visible'):Connect(function()
			if not LibraryMainOuterFrame.Visible then Hovering = false Tip.Visible = false end
		end)
	end
end

-- ─── Highlight ────────────────────────────────────────────────────────────────
function Library:OnHighlight(Trigger, Frame, Props, Leave, Condition)
	local Reg = Library.RegistryMap[Frame]
	local function Apply(P)
		if Condition and not Condition() then return end
		for k,v in next, P do
			Frame[k] = Library[v] or v
			if Reg and Reg.Properties[k] then Reg.Properties[k] = v end
		end
	end
	Trigger.MouseEnter:Connect(function() Apply(Props) end)
	Trigger.MouseMoved:Connect(function() Apply(Props) end)
	Trigger.MouseLeave:Connect(function() Apply(Leave) end)
end

-- ─── Registry & misc ──────────────────────────────────────────────────────────
function Library:MouseIsOverOpenedFrame(Input)
	local Pos = Mouse
	if Library.IsMobile and Input then Pos = Input.Position end
	for Frame in next, Library.OpenedFrames do
		local ap, as = Frame.AbsolutePosition, Frame.AbsoluteSize
		if Pos.X >= ap.X and Pos.X <= ap.X + as.X and Pos.Y >= ap.Y and Pos.Y <= ap.Y + as.Y then
			return true
		end
	end
end

function Library:MouseIsOverFrame(Frame, Input)
	local Pos = Mouse
	if Library.IsMobile and Input then Pos = Input.Position end
	local ap, as = Frame.AbsolutePosition, Frame.AbsoluteSize
	return Pos.X >= ap.X and Pos.X <= ap.X + as.X and Pos.Y >= ap.Y and Pos.Y <= ap.Y + as.Y
end

function Library:UpdateDependencyBoxes()
	for _, d in next, Library.DependencyBoxes do d:Update() end
end

function Library:MapValue(v, a1, a2, b1, b2)
	return (1 - (v-a1)/(a2-a1)) * b1 + ((v-a1)/(a2-a1)) * b2
end

function Library:GetTextBounds(Text, Font, Size, Res)
	local b = TextService:GetTextSize(Text, Size, Font, Res or Vector2.new(1920,1080))
	return b.X, b.Y
end

function Library:GetDarkerColor(c)
	local h,s,v = Color3.toHSV(c)
	return Color3.fromHSV(h, s, v/1.5)
end
Library.AccentColorDark = Library:GetDarkerColor(Library.AccentColor)

function Library:AddToRegistry(Instance, Props, IsHud)
	local d = { Instance = Instance; Properties = Props; Idx = #Library.Registry+1 }
	table.insert(Library.Registry, d)
	Library.RegistryMap[Instance] = d
	if IsHud then table.insert(Library.HudRegistry, d) end
end

function Library:RemoveFromRegistry(Instance)
	local d = Library.RegistryMap[Instance]
	if not d then return end
	for i = #Library.Registry, 1, -1 do
		if Library.Registry[i] == d then table.remove(Library.Registry, i) end
	end
	for i = #Library.HudRegistry, 1, -1 do
		if Library.HudRegistry[i] == d then table.remove(Library.HudRegistry, i) end
	end
	Library.RegistryMap[Instance] = nil
end

function Library:UpdateColorsUsingRegistry()
	for _, obj in next, Library.Registry do
		for prop, idx in next, obj.Properties do
			if type(idx) == 'string' then obj.Instance[prop] = Library[idx]
			elseif type(idx) == 'function' then obj.Instance[prop] = idx() end
		end
	end
end

function Library:GiveSignal(s) table.insert(Library.Signals, s) end

function Library:Unload()
	for i = #Library.Signals, 1, -1 do
		table.remove(Library.Signals, i):Disconnect()
	end
	if Library.OnUnload then Library.OnUnload() end
	ScreenGui:Destroy()
end

function Library:OnUnload(cb) Library.OnUnload = cb end

Library:GiveSignal(ScreenGui.DescendantRemoving:Connect(function(inst)
	if Library.RegistryMap[inst] then Library:RemoveFromRegistry(inst) end
end))

					Swatch.AbsolutePosition.X + Swatch.AbsoluteSize.X + 4,
					Swatch.AbsolutePosition.Y + 1)

				local mw = 60
				for _,l in next, CtxMenu.Inner:GetChildren() do
					if l:IsA('TextLabel') then mw = math.max(mw, l.TextBounds.X) end
				end
				CtxMenu.Container.Size = UDim2.fromOffset(
					mw + 8, CtxMenu.Inner.Layout.AbsoluteContentSize.Y + 4)
			end
			Swatch:GetPropertyChangedSignal('AbsolutePosition'):Connect(updPos)
			CtxMenu.Inner.Layout:GetPropertyChangedSignal('AbsoluteContentSize'):Connect(updSize)
			task.spawn(updPos) task.spawn(updSize)
			Library:AddToRegistry(CtxMenu.Inner, { BackgroundColor3 = 'BackgroundColor'; BorderColor3 = 'OutlineColor' })

			function CtxMenu:Show() if Library.IsMobile then Library.CanDrag=false end self.Container.Visible=true end
			function CtxMenu:Hide() if Library.IsMobile then Library.CanDrag=true end self.Container.Visible=false end
			function CtxMenu:AddOption(Str, Cb)
				if type(Cb) ~= 'function' then Cb = function() end end
				local Btn = Library:CreateLabel({
					Active = false; Size = UDim2.new(1,0,0,15); TextSize = 13; Text = Str;
					ZIndex = 16; Parent = self.Inner; TextXAlignment = Enum.TextXAlignment.Left;
				})
				Library:OnHighlight(Btn, Btn, {TextColor3='AccentColor'}, {TextColor3='FontColor'})
				Btn.InputBegan:Connect(function(I)
					if I.UserInputType ~= Enum.UserInputType.MouseButton1 and I.UserInputType ~= Enum.UserInputType.Touch then return end
					Cb()
				end)
			end
			CtxMenu:AddOption('Copy color', function() Library.ColorClipboard=CP.Value Library:Notify('Copied color!',2) end)
			CtxMenu:AddOption('Paste color', function()
				if not Library.ColorClipboard then return Library:Notify('No color copied!',2) end
				CP:SetValueRGB(Library.ColorClipboard)
			end)
			CtxMenu:AddOption('Copy HEX', function() pcall(setclipboard, CP.Value:ToHex()) Library:Notify('Copied hex!',2) end)
			CtxMenu:AddOption('Copy RGB', function()
				pcall(setclipboard, table.concat({math.floor(CP.Value.R*255),math.floor(CP.Value.G*255),math.floor(CP.Value.B*255)},', '))
				Library:Notify('Copied RGB!',2)
			end)
		end

		Library:AddToRegistry(PanelInner,  { BackgroundColor3='BackgroundColor'; BorderColor3='OutlineColor' })
		Library:AddToRegistry(AccBar,       { BackgroundColor3='AccentColor' })
		Library:AddToRegistry(SatInner,     { BackgroundColor3='BackgroundColor'; BorderColor3='OutlineColor' })
		Library:AddToRegistry(HexInner,     { BackgroundColor3='MainColor'; BorderColor3='OutlineColor' })
		Library:AddToRegistry(RgbOuter.Frame,{BackgroundColor3='MainColor'; BorderColor3='OutlineColor'})
		Library:AddToRegistry(RgbBox,       { TextColor3='FontColor' })
		Library:AddToRegistry(HexBox,       { TextColor3='FontColor' })

		HexBox.FocusLost:Connect(function(enter)
			if enter then
				local ok,r = pcall(Color3.fromHex, HexBox.Text)
				if ok and typeof(r)=='Color3' then CP.Hue,CP.Sat,CP.Vib=Color3.toHSV(r) end
			end
			CP:Display()
		end)
		RgbBox.FocusLost:Connect(function(enter)
			if enter then
				local r,g,b = RgbBox.Text:match('(%d+),%s*(%d+),%s*(%d+)')
				if r then CP.Hue,CP.Sat,CP.Vib=Color3.toHSV(Color3.fromRGB(r,g,b)) end
			end
			CP:Display()
		end)

		function CP:Display()
			CP.Value = Color3.fromHSV(CP.Hue, CP.Sat, CP.Vib)
			SatMap.BackgroundColor3 = Color3.fromHSV(CP.Hue,1,1)
			Library:Create(Swatch, {
				BackgroundColor3=CP.Value; BackgroundTransparency=CP.Transparency;
				BorderColor3=Library:GetDarkerColor(CP.Value);
			})
			if TransInner then
				TransInner.BackgroundColor3 = CP.Value
				TransCursor.Position = UDim2.new(1-CP.Transparency,0,0,0)
			end
			CursorO.Position = UDim2.new(CP.Sat,0,1-CP.Vib,0)
			HueCursor.Position = UDim2.new(0,0,CP.Hue,0)
			HexBox.Text = '#'..CP.Value:ToHex()
			RgbBox.Text = table.concat({math.floor(CP.Value.R*255),math.floor(CP.Value.G*255),math.floor(CP.Value.B*255)},', ')
			Library:SafeCallback(CP.Callback, CP.Value)
			Library:SafeCallback(CP.Changed, CP.Value)
		end

		function CP:OnChanged(f) CP.Changed=f f(CP.Value) end
		if ParentObj.Addons then table.insert(ParentObj.Addons, CP) end
		function CP:Show()
			for f,_ in next, Library.OpenedFrames do
				if f.Name=='Color' then f.Visible=false Library.OpenedFrames[f]=nil end
			end
			PanelOuter.Visible=true Library.OpenedFrames[PanelOuter]=true
		end
		function CP:Hide() PanelOuter.Visible=false Library.OpenedFrames[PanelOuter]=nil end
		function CP:SetValue(hsv,t)
			CP.Transparency=t or 0 CP:SetHSVFromRGB(Color3.fromHSV(hsv[1],hsv[2],hsv[3])) CP:Display()
		end
		function CP:SetValueRGB(c,t) CP.Transparency=t or 0 CP:SetHSVFromRGB(c) CP:Display() end

		SatMap.InputBegan:Connect(function(I)
			if I.UserInputType==Enum.UserInputType.MouseButton1 or I.UserInputType==Enum.UserInputType.Touch then
				while InputService:IsMouseButtonPressed(Enum.UserInputType.MouseButton1) do
					local mn=SatMap.AbsolutePosition local mx=mn+SatMap.AbsoluteSize
					CP.Sat=(math.clamp(Mouse.X,mn.X,mx.X)-mn.X)/(mx.X-mn.X)
					CP.Vib=1-(math.clamp(Mouse.Y,mn.Y,mx.Y)-mn.Y)/(mx.Y-mn.Y)
					CP:Display() RenderStepped:Wait()
				end
				Library:AttemptSave()
			end
		end)
		HueInner.InputBegan:Connect(function(I)
			if I.UserInputType==Enum.UserInputType.MouseButton1 or I.UserInputType==Enum.UserInputType.Touch then
				while InputService:IsMouseButtonPressed(Enum.UserInputType.MouseButton1) do
					local mn=HueInner.AbsolutePosition.Y local mx=mn+HueInner.AbsoluteSize.Y
					CP.Hue=(math.clamp(Mouse.Y,mn,mx)-mn)/(mx-mn)
					CP:Display() RenderStepped:Wait()
				end
				Library:AttemptSave()
			end
		end)
		Swatch.InputBegan:Connect(function(I)
			if Library:MouseIsOverOpenedFrame() then return end
			if I.UserInputType==Enum.UserInputType.MouseButton1 or I.UserInputType==Enum.UserInputType.Touch then
				if PanelOuter.Visible then CP:Hide() else CtxMenu:Hide() CP:Show() end
			elseif I.UserInputType==Enum.UserInputType.MouseButton2 then
				CtxMenu:Show() CP:Hide()
			end
		end)
		if TransInner then
			TransInner.InputBegan:Connect(function(I)
				if I.UserInputType==Enum.UserInputType.MouseButton1 or I.UserInputType==Enum.UserInputType.Touch then
					while InputService:IsMouseButtonPressed(Enum.UserInputType.MouseButton1) do
						local mn=TransInner.AbsolutePosition.X local mx=mn+TransInner.AbsoluteSize.X
						CP.Transparency=1-(math.clamp(Mouse.X,mn,mx)-mn)/(mx-mn)
						CP:Display() RenderStepped:Wait()
					end
					Library:AttemptSave()
				end
			end)
		end
		Library:GiveSignal(InputService.InputBegan:Connect(function(I)
			if I.UserInputType==Enum.UserInputType.MouseButton1 or I.UserInputType==Enum.UserInputType.Touch then
				local ap,as=PanelOuter.AbsolutePosition,PanelOuter.AbsoluteSize
				if Mouse.X<ap.X or Mouse.X>ap.X+as.X or Mouse.Y<ap.Y-21 or Mouse.Y>ap.Y+as.Y then CP:Hide() end
				if not Library:MouseIsOverFrame(CtxMenu.Container) then CtxMenu:Hide() end
			end
			if I.UserInputType==Enum.UserInputType.MouseButton2 and CtxMenu.Container.Visible then
				if not Library:MouseIsOverFrame(CtxMenu.Container) and not Library:MouseIsOverFrame(Swatch) then CtxMenu:Hide() end
			end
		end))

		CP:Display()
		CP.DisplayFrame = Swatch
		Options[Idx] = CP
		return self
	end

	-- ── KeyPicker ────────────────────────────────────────────────────────────
	function Funcs:AddKeyPicker(Idx, Info)
		local ParentObj = self
		local ToggleLabel = self.TextLabel
		assert(Info.Default, 'AddKeyPicker: Missing default value.')

		local KP = {
			Value = Info.Default; Toggled = false;
			Mode = Info.Mode or 'Toggle'; Type = 'KeyPicker';
			Callback = Info.Callback or function() end;
			ChangedCallback = Info.ChangedCallback or function() end;
			SyncToggleState = Info.SyncToggleState or false;
		}
		if KP.SyncToggleState then Info.Modes={'Toggle'} Info.Mode='Toggle' end

		local PickOuter = Library:Create('Frame', {
			BackgroundColor3 = BG3; BorderColor3 = SEP;
			Size = UDim2.new(0,32,0,16); ZIndex = 6; Parent = ToggleLabel;
		})
		local PickInner = Library:Create('Frame', {
			BackgroundColor3 = BG3; BorderColor3 = SEP; BorderMode = Enum.BorderMode.Inset;
			Size = UDim2.new(1,0,1,0); ZIndex = 7; Parent = PickOuter;
		})
		Library:AddToRegistry(PickInner, { BackgroundColor3='MainColor'; BorderColor3='OutlineColor' })
		local Display = Library:CreateLabel({
			Size = UDim2.new(1,0,1,0); TextSize = 12; Text = Info.Default;
			TextWrapped = true; ZIndex = 8; Parent = PickInner;
		})

		local ModeOuter = Library:Create('Frame', {
			BorderColor3 = Color3.new(0,0,0);
			Position = UDim2.fromOffset(ToggleLabel.AbsolutePosition.X+ToggleLabel.AbsoluteSize.X+4, ToggleLabel.AbsolutePosition.Y+1);
			Size = UDim2.new(0,60,0,47); Visible=false; ZIndex=14; Parent=ScreenGui;
		})
		ToggleLabel:GetPropertyChangedSignal('AbsolutePosition'):Connect(function()
			ModeOuter.Position = UDim2.fromOffset(ToggleLabel.AbsolutePosition.X+ToggleLabel.AbsoluteSize.X+4, ToggleLabel.AbsolutePosition.Y+1)
		end)
		local ModeInner = Library:Create('Frame', {
			BackgroundColor3=BG2; BorderColor3=SEP; BorderMode=Enum.BorderMode.Inset;
			Size=UDim2.new(1,0,1,0); ZIndex=15; Parent=ModeOuter;
		})
		Library:AddToRegistry(ModeInner, { BackgroundColor3='BackgroundColor'; BorderColor3='OutlineColor' })
		Library:Create('UIListLayout', { FillDirection=Enum.FillDirection.Vertical; SortOrder=Enum.SortOrder.LayoutOrder; Parent=ModeInner })

		local ContainerLabel = Library:CreateLabel({
			TextXAlignment=Enum.TextXAlignment.Left; Size=UDim2.new(1,0,0,18);
			TextSize=13; Visible=false; ZIndex=110; Parent=Library.KeybindContainer;
		}, true)

		local Modes = Info.Modes or {'Always','Toggle','Hold'}
		local ModeButtons = {}

		for _,Mode in next, Modes do
			local MB = {}
			local Lbl = Library:CreateLabel({
				Active=false; Size=UDim2.new(1,0,0,15); TextSize=13; Text=Mode;
				ZIndex=16; Parent=ModeInner;
			})
			function MB:Select()
				for _,b in next, ModeButtons do b:Deselect() end
				KP.Mode = Mode
				Lbl.TextColor3 = ACC Library.RegistryMap[Lbl].Properties.TextColor3='AccentColor'
				ModeOuter.Visible = false
			end
			function MB:Deselect()
				KP.Mode = nil
				Lbl.TextColor3 = TXT Library.RegistryMap[Lbl].Properties.TextColor3='FontColor'
			end
			Lbl.InputBegan:Connect(function(I)
				if I.UserInputType==Enum.UserInputType.MouseButton1 then MB:Select() Library:AttemptSave() end
			end)
			if Mode == KP.Mode then MB:Select() end
			ModeButtons[Mode] = MB
		end

		function KP:Update()
			if Info.NoUI then return end
			local state = KP:GetState()
			ContainerLabel.Text = string.format('[%s] %s (%s)', KP.Value, Info.Text, KP.Mode)
			ContainerLabel.Visible = true
			ContainerLabel.TextColor3 = state and ACC or TXT
			Library.RegistryMap[ContainerLabel].Properties.TextColor3 = state and 'AccentColor' or 'FontColor'
			local ys,xs = 0,0
			for _,l in next, Library.KeybindContainer:GetChildren() do
				if l:IsA('TextLabel') and l.Visible then
					ys+=18 if l.TextBounds.X>xs then xs=l.TextBounds.X end
				end
			end
			Library.KeybindFrame.Size = UDim2.new(0, math.max(xs+10,210), 0, ys+23)
		end

		function KP:GetState()
			if KP.Mode=='Always' then return true
			elseif KP.Mode=='Hold' then
				if KP.Value=='None' then return false end
				local k=KP.Value
				if k=='MB1' then return InputService:IsMouseButtonPressed(Enum.UserInputType.MouseButton1)
				elseif k=='MB2' then return InputService:IsMouseButtonPressed(Enum.UserInputType.MouseButton2)
				else return InputService:IsKeyDown(Enum.KeyCode[k]) end
			else return KP.Toggled end
		end

		function KP:SetValue(d)
			Display.Text=d[1] KP.Value=d[1]
			if ModeButtons[d[2]] then ModeButtons[d[2]]:Select() end
			KP:Update()
		end
		function KP:OnClick(f) KP.Clicked=f end
		function KP:OnChanged(f) KP.Changed=f f(KP.Value) end
		if ParentObj.Addons then table.insert(ParentObj.Addons, KP) end

		function KP:DoClick()
			if ParentObj.Type=='Toggle' and KP.SyncToggleState then ParentObj:SetValue(not ParentObj.Value) end
			Library:SafeCallback(KP.Callback, KP.Toggled)
			Library:SafeCallback(KP.Clicked, KP.Toggled)
		end

		local Picking = false
		PickOuter.InputBegan:Connect(function(I)
			if I.UserInputType==Enum.UserInputType.MouseButton1 and not Library:MouseIsOverOpenedFrame() then
				Picking = true Display.Text = ''
				local Break, Text = false, ''
				task.spawn(function()
					while not Break do
						if Text=='...' then Text='' end
						Text=Text..'.' Display.Text=Text task.wait(0.4)
					end
				end)
				task.wait(0.2)
				local E E = InputService.InputBegan:Connect(function(I2)
					local k
					if I2.UserInputType==Enum.UserInputType.Keyboard then k=I2.KeyCode.Name
					elseif I2.UserInputType==Enum.UserInputType.MouseButton1 then k='MB1'
					elseif I2.UserInputType==Enum.UserInputType.MouseButton2 then k='MB2' end
					Break=true Picking=false
					Display.Text=k KP.Value=k
					Library:SafeCallback(KP.ChangedCallback, I2.KeyCode or I2.UserInputType)
					Library:SafeCallback(KP.Changed, I2.KeyCode or I2.UserInputType)
					Library:AttemptSave() E:Disconnect()
				end)
			elseif I.UserInputType==Enum.UserInputType.MouseButton2 and not Library:MouseIsOverOpenedFrame() then
				ModeOuter.Visible = true
			end
		end)

		Library:GiveSignal(InputService.InputBegan:Connect(function(I)
			if KP.Value=='Unknown' then return end
			if not Picking and not InputService:GetFocusedTextBox() then
				if KP.Mode=='Toggle' then
					local k=KP.Value
					if k=='MB1' or k=='MB2' then
						if (k=='MB1' and I.UserInputType==Enum.UserInputType.MouseButton1)
							or (k=='MB2' and I.UserInputType==Enum.UserInputType.MouseButton2) then
							KP.Toggled=not KP.Toggled KP:DoClick()
						end
					elseif I.UserInputType==Enum.UserInputType.Keyboard then
						if I.KeyCode.Name==k then KP.Toggled=not KP.Toggled KP:DoClick() end
					end
				end
				KP:Update()
			end
			if I.UserInputType==Enum.UserInputType.MouseButton1 then
				local ap,as=ModeOuter.AbsolutePosition,ModeOuter.AbsoluteSize
				if Mouse.X<ap.X or Mouse.X>ap.X+as.X or Mouse.Y<ap.Y-21 or Mouse.Y>ap.Y+as.Y then
					ModeOuter.Visible=false
				end
			end
		end))
		Library:GiveSignal(InputService.InputEnded:Connect(function()
			if not Picking then KP:Update() end
		end))

		KP:Update()
		KP.DisplayFrame = PickOuter
		Options[Idx] = KP
		return self
	end

	BaseAddons.__index = Funcs
	BaseAddons.__namecall = function(t,k,...) return Funcs[k](...) end
end

	-- Button
	function Funcs:AddButton(...)
		local Button = {}
		local function Parse(obj, ...)
			local p = select(1,...)
			if type(p)=='table' then
				obj.Text=p.Text obj.Func=p.Func obj.DoubleClick=p.DoubleClick obj.Tooltip=p.Tooltip
			else
				obj.Text=select(1,...) obj.Func=select(2,...)
			end
			assert(type(obj.Func)=='function','AddButton: Func missing.')
		end
		Parse(Button, ...)

		local Container = self.Container

		local function MakeBtn(b)
			local Outer = Library:Create('Frame', {
				BackgroundColor3=BG3; BorderColor3=SEP;
				Size=UDim2.new(1,-4,0,22); ZIndex=5;
			})
			local Inner = Library:Create('Frame', {
				BackgroundColor3=BG3; BorderColor3=SEP; BorderMode=Enum.BorderMode.Inset;
				Size=UDim2.new(1,0,1,0); ZIndex=6; Parent=Outer;
			})
			local Lbl = Library:CreateLabel({
				Size=UDim2.new(1,0,1,0); TextSize=13; Text=b.Text; ZIndex=6; Parent=Inner;
			})
			Library:AddToRegistry(Outer, { BorderColor3='OutlineColor'; BackgroundColor3='MainColor' })
			Library:AddToRegistry(Inner, { BackgroundColor3='MainColor'; BorderColor3='OutlineColor' })
			Library:OnHighlight(Outer, Outer,
				{ BackgroundColor3='AccentColor'; BorderColor3='OutlineColor' },
				{ BackgroundColor3='MainColor'; BorderColor3='OutlineColor' })
			return Outer, Inner, Lbl
		end

		local function InitEvents(b)
			local function WFE(ev, t, v)
				local be = Instance.new('BindableEvent')
				local c = ev:Once(function(...)
					if type(v)=='function' and v(...) then be:Fire(true) else be:Fire(false) end
				end)
				task.delay(t, function() c:disconnect() be:Fire(false) end)
				return be.Event:Wait()
			end
			local function Valid(I)
				if Library:MouseIsOverOpenedFrame(I) then return false end
				return I.UserInputType==Enum.UserInputType.MouseButton1
			end

			b.Outer.InputBegan:Connect(function(I)
				if not Valid(I) then return end
				if I.UserInputType==Enum.UserInputType.Touch then return end
				if b.Locked then return end
				if b.DoubleClick then
					Library:RemoveFromRegistry(b.Label)
					Library:AddToRegistry(b.Label, {TextColor3='AccentColor'})
					b.Label.TextColor3=ACC b.Label.Text='Are you sure?' b.Locked=true
					local ok = WFE(b.Outer.InputBegan, 0.5, Valid)
					Library:RemoveFromRegistry(b.Label)
					Library:AddToRegistry(b.Label, {TextColor3='FontColor'})
					b.Label.TextColor3=TXT b.Label.Text=b.Text
					task.defer(rawset, b, 'Locked', false)
					if ok then Library:SafeCallback(b.Func) end
					return
				end
				Library:SafeCallback(b.Func)
			end)
			b.Outer.InputEnded:Connect(function(I)
				if I.UserInputType~=Enum.UserInputType.Touch then return end
				if not Library.TouchIsTap then return end
				if not Library:MouseIsOverFrame(b.Outer, I) then return end
				if b.Locked then return end
				Library:SafeCallback(b.Func)
			end)
		end

		Button.Outer, Button.Inner, Button.Label = MakeBtn(Button)
		Button.Outer.Parent = Container
		InitEvents(Button)

		function Button:AddTooltip(t)
			if type(t)=='string' then Library:AddToolTip(t, self.Outer) end
			return self
		end
		function Button:AddButton(...)
			local Sub = {}
			Parse(Sub, ...)
			self.Outer.Size = UDim2.new(0.5,-2,0,22)
			Sub.Outer, Sub.Inner, Sub.Label = MakeBtn(Sub)
			Sub.Outer.Position = UDim2.new(1,3,0,0)
			Sub.Outer.Size = UDim2.new(1,-3,1,0)
			Sub.Outer.Parent = self.Outer
			function Sub:AddTooltip(t)
				if type(t)=='string' then Library:AddToolTip(t, self.Outer) end
				return Sub
			end
			if type(Sub.Tooltip)=='string' then Sub:AddTooltip(Sub.Tooltip) end
			InitEvents(Sub)
			return Sub
		end

		if type(Button.Tooltip)=='string' then Button:AddTooltip(Button.Tooltip) end
		self:AddBlank(5) self:Resize()
		return Button
	end

			BackgroundTransparency=1; ClipsDescendants=true;
			Position=UDim2.new(0,5,0,0); Size=UDim2.new(1,-5,1,0);
			ZIndex=7; Parent=Inner;
		})

			BackgroundTransparency=1; Position=UDim2.fromOffset(0,0); Size=UDim2.fromScale(5,1);
			Font=Library.Font; PlaceholderColor3=TXT2; PlaceholderText=Info.Placeholder or '';
			Text=Info.Default or ''; TextColor3=TXT; TextSize=13; TextStrokeTransparency=0;
			TextXAlignment=Enum.TextXAlignment.Left;
			ClearTextOnFocus=(typeof(Info.ClearTextOnFocus)~='boolean' and true or Info.ClearTextOnFocus);
			ZIndex=7; Parent=Clip;
		})
		Library:ApplyTextStroke(Box)

		function TB:SetValue(t)
			if Info.MaxLength and #t>Info.MaxLength then t=t:sub(1,Info.MaxLength) end
			if TB.Numeric and not tonumber(t) and #t>0 then t=TB.Value end
			TB.Value=t Box.Text=t
			Library:SafeCallback(TB.Callback, TB.Value)
			Library:SafeCallback(TB.Changed, TB.Value)
		end

		if TB.Finished then
			Box.FocusLost:Connect(function(enter) if enter then TB:SetValue(Box.Text) Library:AttemptSave() end end)
		else
			Box:GetPropertyChangedSignal('Text'):Connect(function() TB:SetValue(Box.Text) Library:AttemptSave() end)
		end

		local function Upd()
			local PAD=2 local rev=Clip.AbsoluteSize.X
			if not Box:IsFocused() or Box.TextBounds.X<=rev-2*PAD then
				Box.Position=UDim2.new(0,PAD,0,0)
			else
				local cur=Box.CursorPosition
				if cur~=-1 then
					local w=TextService:GetTextSize(Box.Text:sub(1,cur-1),Box.TextSize,Box.Font,Vector2.new(math.huge,math.huge)).X
					local pos=Box.Position.X.Offset+w
					if pos<PAD then Box.Position=UDim2.fromOffset(PAD-w,0)
					elseif pos>rev-PAD-1 then Box.Position=UDim2.fromOffset(rev-w-PAD-1,0) end
				end
			end
		end

		Box:GetPropertyChangedSignal('Text'):Connect(Upd)
		Box:GetPropertyChangedSignal('CursorPosition'):Connect(Upd)
		Box.FocusLost:Connect(Upd) Box.Focused:Connect(Upd)
		Library:AddToRegistry(Box, { TextColor3='FontColor' })

		function TB:OnChanged(f) TB.Changed=f f(TB.Value) end
		self:AddBlank(5) self:Resize()
		Options[Idx] = TB
		return TB
	end

	-- Slider
	function Funcs:AddSlider(Idx, Info)
		assert(Info.Default,'AddSlider: Default missing.')
		assert(Info.Text,'AddSlider: Text missing.')
		assert(Info.Min,'AddSlider: Min missing.')
		assert(Info.Max,'AddSlider: Max missing.')
		assert(Info.Rounding,'AddSlider: Rounding missing.')

		local Slider = {
			Value=Info.Default; Min=Info.Min; Max=Info.Max;
			Rounding=Info.Rounding; MaxSize=232; Type='Slider';
			Callback=Info.Callback or function() end;
		}
		local Container = self.Container

		if not Info.Compact then
			Library:CreateLabel({
				Size=UDim2.new(1,0,0,12); TextSize=13; Text=Info.Text;
				TextXAlignment=Enum.TextXAlignment.Left; TextYAlignment=Enum.TextYAlignment.Bottom;
				ZIndex=5; Parent=Container;
			})
			self:AddBlank(3)
		end

		local Track = Library:Create('Frame', {
			BackgroundColor3=BG3; BorderColor3=SEP;
			Size=UDim2.new(1,-4,0,8); ZIndex=5; Parent=Container;
		})
		Track:GetPropertyChangedSignal('AbsoluteSize'):Connect(function()
			Slider.MaxSize = Track.AbsoluteSize.X - 2
		end)
		Library:AddToRegistry(Track, { BackgroundColor3='MainColor'; BorderColor3='OutlineColor' })

		local Fill = Library:Create('Frame', {
			BackgroundColor3=ACC; BorderSizePixel=0;
			Size=UDim2.new(0,0,1,0); ZIndex=6; Parent=Track;
		})
		Library:AddToRegistry(Fill, { BackgroundColor3='AccentColor' })

		local Thumb = Library:Create('Frame', {
			BackgroundColor3=Color3.new(1,1,1); BorderSizePixel=0;
			AnchorPoint=Vector2.new(0.5,0.5); Position=UDim2.new(0,0,0.5,0);
			Size=UDim2.new(0,10,0,10); ZIndex=8; Parent=Track;
		})
		Library:Create('UICorner', { CornerRadius=UDim.new(1,0); Parent=Thumb })

		local ValLabel = Library:CreateLabel({
			Size=UDim2.new(1,0,0,12); TextSize=12; Text='';
			ZIndex=7; Parent=Container;
		})

		Library:OnHighlight(Track, Track, {BorderColor3='AccentColor'}, {BorderColor3='OutlineColor'})
		if type(Info.Tooltip)=='string' then Library:AddToolTip(Info.Tooltip, Track) end

		function Slider:UpdateColors()
			Fill.BackgroundColor3=ACC Fill.BorderColor3=ACC2
		end

		function Slider:Display()
			local suf = Info.Suffix or ''
			if Info.Compact then ValLabel.Text=Info.Text..': '..Slider.Value..suf
			elseif Info.HideMax then ValLabel.Text=tostring(Slider.Value)..suf
			else ValLabel.Text=tostring(Slider.Value)..suf..' / '..tostring(Slider.Max)..suf end

			local x = Library:MapValue(Slider.Value, Slider.Min, Slider.Max, 0, 1)
			Fill.Size = UDim2.new(x,0,1,0)
			Thumb.Position = UDim2.new(x,0,0.5,0)
			Thumb.Visible = x > 0
		end

		function Slider:OnChanged(f) Slider.Changed=f f(Slider.Value) end

		local function Round(v)
			if Slider.Rounding==0 then return math.floor(v) end
			return tonumber(string.format('%.'..Slider.Rounding..'f', v))
		end

		function Slider:GetValueFromXScale(x)
			return Round(Library:MapValue(x, 0, 1, Slider.Min, Slider.Max))
		end

		function Slider:SetMax(v)
			assert(v>Slider.Min,'Max <= Min')
			Slider.Value=math.clamp(Slider.Value,Slider.Min,v) Slider.Max=v Slider:Display()
		end
		function Slider:SetMin(v)
			assert(v<Slider.Max,'Min >= Max')
			Slider.Value=math.clamp(Slider.Value,v,Slider.Max) Slider.Min=v Slider:Display()
		end
		function Slider:SetValue(s)
			local n=tonumber(s) if not n then return end
			n=math.clamp(n,Slider.Min,Slider.Max) Slider.Value=n Slider:Display()
			Library:SafeCallback(Slider.Callback, Slider.Value)
			Library:SafeCallback(Slider.Changed, Slider.Value)
		end

		-- mouse
		Track.InputBegan:Connect(function(I)
			if I.UserInputType==Enum.UserInputType.MouseButton1 and not Library:MouseIsOverOpenedFrame() then
				local Sides={}
				if Library.Window then Sides=Library.Window.Tabs[Library.ActiveTab]:GetSides() end
				for _,s in pairs(Sides) do
					if typeof(s)=='Instance' and s:IsA('ScrollingFrame') then s.ScrollingEnabled=false end
				end
				local mp=Mouse.X local gp=Fill.AbsoluteSize.X
				local df=mp-(Fill.AbsolutePosition.X+gp)
				while InputService:IsMouseButtonPressed(Enum.UserInputType.MouseButton1) do
					local nX=math.clamp(gp+(Mouse.X-mp)+df,0,Slider.MaxSize)
					local nV=Slider:GetValueFromXScale(Library:MapValue(nX,0,Slider.MaxSize,0,1))
					local old=Slider.Value Slider.Value=nV Slider:Display()
					if nV~=old then Library:SafeCallback(Slider.Callback,nV) Library:SafeCallback(Slider.Changed,nV) end
					RenderStepped:Wait()
				end
				for _,s in pairs(Sides) do
					if typeof(s)=='Instance' and s:IsA('ScrollingFrame') then s.ScrollingEnabled=true end
				end
				Library:AttemptSave()
			end
		end)

		-- touch
		if Library.IsMobile then
			InputService.TouchStarted:Connect(function(I)
				if not Library:MouseIsOverFrame(Track, I) then return end
				if not Library.Window or not Library.Window.Holder.Visible then return end

				local startPos=I.Position local startX=I.Position.X
				local gp=Fill.AbsoluteSize.X
				local df=startX-(Fill.AbsolutePosition.X+gp)
				local Dir, Committed = nil, false
				local Sides={}
				if Library.Window then Sides=Library.Window.Tabs[Library.ActiveTab]:GetSides() end

				local MC, EC
				MC = InputService.TouchMoved:Connect(function(T)
					if not Committed then
						local d=T.Position-startPos
						local ax,ay=math.abs(d.X),math.abs(d.Y)
						if ax<8 and ay<8 then return end
						Dir = ax > ay*2
						Committed = true
						if Dir then
							Library.CanDrag=false
							for _,s in pairs(Sides) do
								if typeof(s)=='Instance' and s:IsA('ScrollingFrame') then s.ScrollingEnabled=false end
							end
						else
							MC:Disconnect() EC:Disconnect()
						end
						return
					end
					if not Dir then return end
					local nX=math.clamp(gp+(T.Position.X-startX)+df,0,Slider.MaxSize)
					local nV=Slider:GetValueFromXScale(Library:MapValue(nX,0,Slider.MaxSize,0,1))
					local old=Slider.Value Slider.Value=nV Slider:Display()
					if nV~=old then Library:SafeCallback(Slider.Callback,nV) Library:SafeCallback(Slider.Changed,nV) end
				end)
				EC = InputService.TouchEnded:Connect(function()
					MC:Disconnect() EC:Disconnect()
					Library.CanDrag=true
					for _,s in pairs(Sides) do
						if typeof(s)=='Instance' and s:IsA('ScrollingFrame') then s.ScrollingEnabled=true end
					end
					if Dir then Library:AttemptSave() end
				end)
			end)
		end

		Slider:Display()
		self:AddBlank(Info.BlankSize or 6) self:Resize()
		Options[Idx] = Slider
		return Slider
	end

			local Count=0

					else

		end)
		InputService.InputBegan:Connect(function(I)
			if I.UserInputType==Enum.UserInputType.MouseButton1 or I.UserInputType==Enum.UserInputType.Touch then
				local ap,as=ListOuter.AbsolutePosition,ListOuter.AbsoluteSize
				if Mouse.X<ap.X or Mouse.X>ap.X+as.X or Mouse.Y<ap.Y-21 or Mouse.Y>ap.Y+as.Y then
					DD:CloseDropdown()
				end
			end
		end)

		DD:BuildDropdownList() DD:Display()

		-- defaults
		local defs={}
		if type(Info.Default)=='string' then
			local i=table.find(DD.Values,Info.Default) if i then table.insert(defs,i) end
		elseif type(Info.Default)=='table' then
			for _,v in next, Info.Default do local i=table.find(DD.Values,v) if i then table.insert(defs,i) end end
		elseif type(Info.Default)=='number' and DD.Values[Info.Default] then
			table.insert(defs,Info.Default)
		end
		if next(defs) then
			for _,i in ipairs(defs) do
				if Info.Multi then DD.Value[DD.Values[i]]=true else DD.Value=DD.Values[i] end
				if not Info.Multi then break end
			end
			DD:BuildDropdownList() DD:Display()
		end

		self:AddBlank(Info.BlankSize or 5) self:Resize()
		Options[Idx] = DD
		return DD
	end

	-- DependencyBox
	function Funcs:AddDependencyBox()
		local Dep = { Dependencies={} }
		local Container = self.Container
		local Holder = Library:Create('Frame', {
			BackgroundTransparency=1; Size=UDim2.new(1,0,0,0); Visible=false; Parent=Container;
		})
		local Frame = Library:Create('Frame', {
			BackgroundTransparency=1; Size=UDim2.new(1,0,1,0); Parent=Holder;
		})
		local Layout = Library:Create('UIListLayout', {
			FillDirection=Enum.FillDirection.Vertical; SortOrder=Enum.SortOrder.LayoutOrder; Parent=Frame;
		})
		local gb = self
		function Dep:Resize() Holder.Size=UDim2.new(1,0,0,Layout.AbsoluteContentSize.Y) gb:Resize() end
		Layout:GetPropertyChangedSignal('AbsoluteContentSize'):Connect(function() Dep:Resize() end)
		Holder:GetPropertyChangedSignal('Visible'):Connect(function() Dep:Resize() end)
		function Dep:Update()
			for _,d in next, Dep.Dependencies do
				local elem,val=d[1],d[2]
				if if elem.Multi then not table.find(elem:GetActiveValues(),val) else elem.Value~=val then
					Holder.Visible=false Dep:Resize() return
				end
			end
			Holder.Visible=true Dep:Resize()
		end
		function Dep:SetupDependencies(deps)
			for _,d in next, deps do
				assert(type(d)=='table','SetupDependencies: not table')
				assert(d[1],'SetupDependencies: missing element')
				assert(d[2]~=nil,'SetupDependencies: missing value')
			end
			Dep.Dependencies=deps Dep:Update()
		end
		Dep.Container=Frame
		setmetatable(Dep, BaseGroupbox)
		table.insert(Library.DependencyBoxes, Dep)
		return Dep
	end

	BaseGroupbox.__index = Funcs
	BaseGroupbox.__namecall = function(t,k,...) return Funcs[k](...) end
end

					TextTransparency=0.4; ZIndex=6; Parent=TBtn;
				})
				local TAccBar = Library:Create('Frame', {
					BackgroundColor3=ACC; BorderSizePixel=0;
					AnchorPoint=Vector2.new(0,1); Position=UDim2.new(0,2,1,-1);
					Size=UDim2.new(1,-4,0,2); BackgroundTransparency=1; ZIndex=6; Parent=TBtn;
				})
				Library:AddToRegistry(TAccBar, { BackgroundColor3='AccentColor' })

				local Container = Library:Create('Frame', {
					BackgroundTransparency=1; Size=UDim2.new(1,0,1,0);
					Visible=false; ZIndex=4; Parent=TabboxContent;
				})
				Library:Create('UIListLayout', {
					FillDirection=Enum.FillDirection.Vertical;
					SortOrder=Enum.SortOrder.LayoutOrder; Parent=Container;
				})

				function Tab:Show()
					for _,t in next, Tabbox.Tabs do t:Hide() end
					TBtn.BackgroundColor3=BG3
					Library.RegistryMap[TBtn].Properties.BackgroundColor3='MainColor'
					TAccBar.BackgroundTransparency=0
					TLbl.Font=Enum.Font.GothamBold TLbl.TextTransparency=0
					Container.Visible=true
					Tab:Resize()
				end
				function Tab:Hide()
					TBtn.BackgroundColor3=BG2
					Library.RegistryMap[TBtn].Properties.BackgroundColor3='BackgroundColor'
					TAccBar.BackgroundTransparency=1
					TLbl.Font=Library.Font TLbl.TextTransparency=0.4
					Container.Visible=false
				end
				function Tab:Resize()
					local s=0
					for _,e in next, Container:GetChildren() do
						if not e:IsA('UIListLayout') and e.Visible then s+=e.Size.Y.Offset end
					end
					BoxOuter.Size=UDim2.new(1,0,0,20+s+4)
				end

				TBtn.InputBegan:Connect(function(I)
					if I.UserInputType==Enum.UserInputType.MouseButton1 and not Library:MouseIsOverOpenedFrame() then
						Tab:Show() Tab:Resize()
					end
				end)
				TBtn.InputEnded:Connect(function(I)
					if I.UserInputType==Enum.UserInputType.Touch then
						if not Library.TouchIsTap then return end
						if not Library:MouseIsOverFrame(TBtn,I) then return end
						Tab:Show() Tab:Resize()
					end
				end)

				Tab.Container=Container
				Tabbox.Tabs[Name]=Tab
				setmetatable(Tab, BaseGroupbox)
				Tab:AddBlank(3) Tab:Resize()

				local count=0
				for _ in next, Tabbox.Tabs do count+=1 end
				if count==1 then Tab:Show() end

				return Tab
			end

			Tab.Tabboxes[Info.Name or '']=Tabbox
			return Tabbox
		end
		function Tab:AddLeftTabbox(n) return Tab:AddTabbox({Name=n,Side=1}) end
		function Tab:AddRightTabbox(n) return Tab:AddTabbox({Name=n,Side=2}) end

		-- tab input events
		TabBtn.InputBegan:Connect(function(I)
			if I.UserInputType==Enum.UserInputType.MouseButton1 then Tab:ShowTab() end
		end)
		TabBtn.InputEnded:Connect(function(I)
			if I.UserInputType==Enum.UserInputType.Touch then
				if not Library.TouchIsTap then return end
				if not Library:MouseIsOverFrame(TabBtn,I) then return end
				Tab:ShowTab()
			end
		end)

		Library.TotalTabs+=1
		if Library.TotalTabs==1 then Tab:ShowTab() end

		Window.Tabs[Name]=Tab
		return Tab
	end

	-- modal element for focus
	local Modal = Library:Create('TextButton', {
		BackgroundTransparency=1; Size=UDim2.new(0,0,0,0);
		Visible=true; Text=''; Modal=false; Parent=ScreenGui;
	})

	-- fade + toggle logic
	local Toggled = false
	local Fading  = false
	local TransCache = {}

	function Library:Toggle()
		if Fading then return end
		local FT=Cfg.MenuFadeTime
		Fading=true Toggled=not Toggled Library.Toggled=Toggled
		Modal.Modal=Toggled

		if Toggled then
			Outer.Visible=true
			if Library.ShowCustomCursor and Drawing then
				local Cur=Drawing.new('Triangle') Cur.Thickness=1 Cur.Filled=true Cur.Visible=true
				local CurO=Drawing.new('Triangle') CurO.Thickness=1 CurO.Filled=false CurO.Color=Color3.new(0,0,0) CurO.Visible=true
				local OldState=InputService.MouseIconEnabled
				RunService:BindToRenderStep('LinoriaCursor',Enum.RenderPriority.Camera.Value-1,function()
					InputService.MouseIconEnabled=false
					local mp=InputService:GetMouseLocation() local x,y=mp.X,mp.Y
					Cur.Color=Library.AccentColor
					Cur.PointA=Vector2.new(x,y) Cur.PointB=Vector2.new(x+16,y+6) Cur.PointC=Vector2.new(x+6,y+16)
					CurO.PointA=Cur.PointA CurO.PointB=Cur.PointB CurO.PointC=Cur.PointC
					if not (Toggled and ScreenGui.Parent and Library.ShowCustomCursor) then
						InputService.MouseIconEnabled=OldState
						Cur:Destroy() CurO:Destroy()
						RunService:UnbindFromRenderStep('LinoriaCursor')
					end
				end)
			end
		end

		for _,opt in Options do
			if opt.Type=='Dropdown' then opt:CloseDropdown() end
		end

		for _,desc in next, Outer:GetDescendants() do
			local props={}
			if desc:IsA('ImageLabel') then table.insert(props,'ImageTransparency') table.insert(props,'BackgroundTransparency')
			elseif desc:IsA('TextLabel') or desc:IsA('TextBox') then table.insert(props,'TextTransparency')
			elseif desc:IsA('Frame') or desc:IsA('ScrollingFrame') then table.insert(props,'BackgroundTransparency')
			elseif desc:IsA('UIStroke') then table.insert(props,'Transparency') end
			local cache=TransCache[desc]
			if not cache then cache={} TransCache[desc]=cache end
			for _,prop in next, props do
				if not cache[prop] then cache[prop]=desc[prop] end
				if cache[prop]==1 then continue end
				TweenService:Create(desc,TweenInfo.new(FT,Enum.EasingStyle.Linear),{[prop]=Toggled and cache[prop] or 1}):Play()
			end
		end
		task.wait(FT) Outer.Visible=Toggled Fading=false
	end

	Library:GiveSignal(InputService.InputBegan:Connect(function(I, Processed)
		if type(Library.ToggleKeybind)=='table' and Library.ToggleKeybind.Type=='KeyPicker' then
			if I.UserInputType==Enum.UserInputType.Keyboard and I.KeyCode.Name==Library.ToggleKeybind.Value then
				task.spawn(Library.Toggle)
			end
		elseif I.KeyCode==Enum.KeyCode.RightControl or (I.KeyCode==Enum.KeyCode.RightShift and not Processed) then
			task.spawn(Library.Toggle)
		end
	end))

	-- Mobile buttons
	if Library.IsMobile then
		-- Toggle UI button
		local TUIOuter = Library:Create('Frame', {
			BorderColor3=SEP; Position=UDim2.new(0.008,0,0.018,0);
			Size=UDim2.new(0,84,0,28); ZIndex=200; Visible=true; Parent=ScreenGui;
		})
		local TUIInner = Library:Create('Frame', {
			BackgroundColor3=BG2; BorderColor3=ACC; BorderMode=Enum.BorderMode.Inset;
			Size=UDim2.new(1,0,1,0); ZIndex=201; Parent=TUIOuter;
		})
		Library:AddToRegistry(TUIInner, { BorderColor3='AccentColor' })
		local TUIBar = Library:Create('Frame', {
			BackgroundColor3=ACC; BorderSizePixel=0; Size=UDim2.new(0,2,1,0); ZIndex=202; Parent=TUIInner;
		})
		Library:AddToRegistry(TUIBar, { BackgroundColor3='AccentColor' })
		local TUIBtn = Library:Create('TextButton', {
			Position=UDim2.new(0,6,0,0); Size=UDim2.new(1,-6,1,0);
			BackgroundTransparency=1; Font=Library.Font; Text='Toggle UI';
			TextColor3=TXT; TextSize=13; TextStrokeTransparency=0;
			ZIndex=203; Parent=TUIInner;
		})
		Library:ApplyTextStroke(TUIBtn)
		Library:MakeDraggable(TUIOuter)
		TUIBtn.MouseButton1Down:Connect(function() task.spawn(Library.Toggle) end)

		-- Lock UI button
		local LUIOuter = Library:Create('Frame', {
			BorderColor3=SEP; Position=UDim2.new(0.008,0,0.075,0);
			Size=UDim2.new(0,84,0,28); ZIndex=200; Visible=true; Parent=ScreenGui;
		})
		local LUIInner = Library:Create('Frame', {
			BackgroundColor3=BG2; BorderColor3=ACC; BorderMode=Enum.BorderMode.Inset;
			Size=UDim2.new(1,0,1,0); ZIndex=201; Parent=LUIOuter;
		})
		Library:AddToRegistry(LUIInner, { BorderColor3='AccentColor' })
		local LUIBar = Library:Create('Frame', {
			BackgroundColor3=ACC; BorderSizePixel=0; Size=UDim2.new(0,2,1,0); ZIndex=202; Parent=LUIInner;
		})
		Library:AddToRegistry(LUIBar, { BackgroundColor3='AccentColor' })
		local LUIBtn = Library:Create('TextButton', {
			Position=UDim2.new(0,6,0,0); Size=UDim2.new(1,-6,1,0);
			BackgroundTransparency=1; Font=Library.Font; Text='Lock UI';
			TextColor3=TXT; TextSize=13; TextStrokeTransparency=0;
			ZIndex=203; Parent=LUIInner;
		})
		Library:ApplyTextStroke(LUIBtn)
		Library:MakeDraggable(LUIOuter)
		LUIBtn.MouseButton1Down:Connect(function()
			Library.CantDragForced=not Library.CantDragForced
			LUIBtn.Text=Library.CantDragForced and 'Unlock UI' or 'Lock UI'
		end)
	end

	if Cfg.AutoShow then task.spawn(Library.Toggle) end
	Window.Holder = Outer
	Library.Window = Window
	return Window
end

getgenv().Library = Library
return Library
